rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Collection users: hanya user yang terautentikasi yang bisa baca/tulis data mereka sendiri
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Collection presensi: user bisa baca data mereka sendiri, admin bisa baca semua
    match /presensi/{docId} {
      allow read: if request.auth != null && 
        (resource.data.uid == request.auth.uid || exists(/databases/$(database)/documents/users/$(request.auth.uid)/roleAdmin));
      allow write: if request.auth != null && request.auth.uid == resource.data.uid;
    }
    
    // Collection notifikasi: user bisa baca notifikasi mereka sendiri, admin bisa baca semua
    match /notifikasi/{docId} {
      allow read: if request.auth != null && 
        (resource.data.targetUid == request.auth.uid || resource.data.targetRole == 'admin' || exists(/databases/$(database)/documents/users/$(request.auth.uid)/roleAdmin));
      allow write: if request.auth != null;
    }
    
    // Collection cuti: karyawan bisa ajukan cuti, admin bisa baca semua
    match /cuti/{docId} {
      allow read: if request.auth != null && 
        (resource.data.uid == request.auth.uid || exists(/databases/$(database)/documents/users/$(request.auth.uid)/roleAdmin));
      allow write: if request.auth != null;
    }
  }
}
