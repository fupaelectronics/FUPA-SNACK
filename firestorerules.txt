rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Koleksi users: hanya user yang login dapat membaca/menulis data mereka sendiri
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      
      // Admin dapat membaca semua data user
      allow read: if request.auth != null && 
        (request.auth.uid == 'O1SJ7hYop3UJjDcsA3JqT29aapI3' || 
         request.auth.uid == 'uB2XsyM6fXUj493cRlHCqpe2fxH3');
    }
    
    // Koleksi presensi: user dapat menambah dan membaca data mereka sendiri
    match /presensi/{document} {
      allow create: if request.auth != null;
      allow read: if request.auth != null && 
        (resource.data.uid == request.auth.uid ||
         request.auth.uid == 'O1SJ7hYop3UJjDcsA3JqT29aapI3' || 
         request.auth.uid == 'uB2XsyM6fXUj493cRlHCqpe2fxH3');
    }
    
    // Koleksi cuti: user dapat menambah dan membaca data mereka sendiri
    match /cuti/{document} {
      allow create: if request.auth != null;
      allow read, update: if request.auth != null && 
        (resource.data.uid == request.auth.uid ||
         request.auth.uid == 'O1SJ7hYop3UJjDcsA3JqT29aapI3' || 
         request.auth.uid == 'uB2XsyM6fXUj493cRlHCqpe2fxH3');
    }
    
    // Koleksi notifikasi: user dapat membaca notifikasi mereka sendiri
    match /notifikasi/{document} {
      allow read: if request.auth != null && 
        (resource.data.targetUid == request.auth.uid ||
         resource.data.targetRole == 'all' ||
         request.auth.uid == 'O1SJ7hYop3UJjDcsA3JqT29aapI3' || 
         request.auth.uid == 'uB2XsyM6fXUj493cRlHCqpe2fxH3');
      allow create: if request.auth != null;
      allow delete: if request.auth != null && request.auth.uid == resource.data.targetUid;
    }
  }
}
